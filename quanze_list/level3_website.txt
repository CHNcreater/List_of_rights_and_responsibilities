function getParam(paramName) {
    paramValue = "", isFound = !1; 
    if (this.location.search.indexOf("?") == 0 && this.location.search.indexOf("=") > 1) { 
        arrSource = decodeURI(this.location.search).substring(1, this.location.search.length).split("&"), i = 0; 
        while (i < arrSource.length && !isFound) arrSource[i].indexOf("=") > 0 && arrSource[i].split("=")[0].toLowerCase() == paramName.toLowerCase() && (paramValue = arrSource[i].split("=")[1], isFound = !0), i++ 
    } 
    if(paramValue==null || paramValue==""){
       return paramValue  == "" && (paramValue  = null), paramValue
    }else{
       return removeXss(paramValue);
    }
    //return paramValue  == "" && (paramValue  = null), paramValue?removeXss(paramValue):null
}

function changeURLArg(url,arg,arg_val){ 
   var pattern=arg+'=([^&]*)'; 
   var replaceText=arg+'='+arg_val; 
   if(url.match(pattern)){ 
       var tmp='/('+ arg+'=)([^&]*)/gi'; 
       tmp=url.replace(eval(tmp),replaceText); 
       return tmp; 
   }else{ 
       if(url.match('[\?]')){ 
           return url+'&'+replaceText; 
       }else{ 
           return url+'?'+replaceText; 
       } 
  } 
   return url+'\n'+arg+'\n'+arg_val; 
}

function navhd(grbsnav,item,spantag){
	// 鼠标放在哪个li上面，span对应一道到哪里，移开后回到原位置
    var grbsnav=document.getElementById(grbsnav);
    var liArr = grbsnav.getElementsByClassName(item);
    var liWidth = liArr[0].offsetWidth;
    var span = grbsnav.getElementsByClassName(spantag)[0];
    // 计数器
    var cnt = 0;

    // for循环绑定事件
    for(var i=0; i<liArr.length; i++){
        // 自定义属性，然后绑定index属性为索引值
        liArr[i].index = i;
        // 鼠标进入事件
        liArr[i].onmouseover = function(){
            // 然span运动到该li的索引值位置
            animate(span, this.index*liWidth);
            cnt = this.index;
            animate(span, cnt*liWidth);
        }
        // 鼠标移开
        liArr[i].onmouseout = function(){
            // span运动到原位置
            animate(span, cnt*liWidth);
        }
    }
}

// 缓动动画封装
function animate(element, target){
    // 清除间歇调用
    clearInterval(element.timer);
    // 设置超时调用
    element.timer = setInterval(function(){
        // 设置步数
        var step = (target - element.offsetLeft)/10;
        // 调整步数
        step = step > 0 ? Math.ceil(step) : Math.floor(step);
        // 设置样式
        element.style.left = element.offsetLeft + step + "px";
        // console.log(1);
        if(Math.abs(target - element.offsetLeft) < Math.abs(step)){
            element.style.left = target + "px";
            clearInterval(element.timer);
        }
    }, 20);
}

//去掉特殊字符
function checkName(val){ 
    var reg = new RegExp("[`~!@#$^&*()=|{}':;',\\[\\].<>/?~！@#￥……&*（）——|{}【】‘；：”“'。，、？]"); 
    var rs = ""; 
    for (var i = 0, l = val.length; i < val.length; i++) { 
        rs = rs + val.substr(i, 1).replace(reg, ''); 
    } 
    return rs; 
}

		}
	}

function openlink(url){
	if(url==""){
	  alert("未发现链接地址");
	  return false;
	}
  if(url.indexOf("zwfw.guizhou.gov.cn")!=-1||url.indexOf(".do")!=-1||url.indexOf("eptemp.aspx")!=-1){
		var qycode=getParam("areacode");
		if(qycode==""||qycode==null||qycode=="null"||!qycode){
			qycode="520000";
		}
		if(url.indexOf("?") == -1){
			location.href=url+'?areacode='+qycode;
		}else{
			location.href=url+'&areacode='+qycode;
		}
	}else{
		location.href=url
	}
}

 * 调用方法:getParam("name") 
 * 返回值:tyler

function getParam(paramName) {
    paramValue = "", isFound = !1; 
    if (this.location.search.indexOf("?") == 0 && this.location.search.indexOf("=") > 1) { 
        arrSource = decodeURI(this.location.search).substring(1, this.location.search.length).split("&"), i = 0; 
        while (i < arrSource.length && !isFound) arrSource[i].indexOf("=") > 0 && arrSource[i].split("=")[0].toLowerCase() == paramName.toLowerCase() && (paramValue = arrSource[i].split("=")[1], isFound = !0), i++ 
    } 
    if(paramValue==null || paramValue==""){
       return paramValue  == "" && (paramValue  = null), paramValue
    }else{
       return removeXss(paramValue);
    }
    //return paramValue  == "" && (paramValue  = null), paramValue?removeXss(paramValue):null
}

            var re = new RegExp('(&#[x|X]0{0,8}'+ parseInt(search[i].charCodeAt(),16)+';?)','gi');
            val = val.replace(re, search[i]);
            re = new RegExp('(&#0{0,8}'+ search[i].charCodeAt() + ';?)','gi');

function changeURLArg(url,arg,arg_val){ 
   var pattern=arg+'=([^&]*)'; 
   var replaceText=arg+'='+arg_val; 
   if(url.match(pattern)){ 
       var tmp='/('+ arg+'=)([^&]*)/gi'; 
       tmp=url.replace(eval(tmp),replaceText); 
       return tmp; 
   }else{ 
       if(url.match('[\?]')){ 
           return url+'&'+replaceText; 
       }else{ 
           return url+'?'+replaceText; 
       } 
  } 
   return url+'\n'+arg+'\n'+arg_val; 
}

function navhd(grbsnav,item,spantag){
	// 鼠标放在哪个li上面，span对应一道到哪里，移开后回到原位置
    var grbsnav=document.getElementById(grbsnav);
    var liArr = grbsnav.getElementsByClassName(item);
    var liWidth = liArr[0].offsetWidth;
    var span = grbsnav.getElementsByClassName(spantag)[0];
    // 计数器
    var cnt = 0;

    // for循环绑定事件
    for(var i=0; i<liArr.length; i++){
        // 自定义属性，然后绑定index属性为索引值
        liArr[i].index = i;
        // 鼠标进入事件
        liArr[i].onmouseover = function(){
            // 然span运动到该li的索引值位置
            animate(span, this.index*liWidth);
            cnt = this.index;
            animate(span, cnt*liWidth);
        }
        // 鼠标移开
        liArr[i].onmouseout = function(){
            // span运动到原位置
            animate(span, cnt*liWidth);
        }
    }
}

// 缓动动画封装
function animate(element, target){
    // 清除间歇调用
    clearInterval(element.timer);
    // 设置超时调用
    element.timer = setInterval(function(){
        // 设置步数
        var step = (target - element.offsetLeft)/10;
        // 调整步数
        step = step > 0 ? Math.ceil(step) : Math.floor(step);
        // 设置样式
        element.style.left = element.offsetLeft + step + "px";
        // console.log(1);
        if(Math.abs(target - element.offsetLeft) < Math.abs(step)){
            element.style.left = target + "px";
            clearInterval(element.timer);
        }
    }, 20);
}

function getParam(paramName) {
    paramValue = "", isFound = !1; 
    if (this.location.search.indexOf("?") == 0 && this.location.search.indexOf("=") > 1) { 
        arrSource = decodeURI(this.location.search).substring(1, this.location.search.length).split("&"), i = 0; 
        while (i < arrSource.length && !isFound) arrSource[i].indexOf("=") > 0 && arrSource[i].split("=")[0].toLowerCase() == paramName.toLowerCase() && (paramValue = arrSource[i].split("=")[1], isFound = !0), i++ 
    } 
    if(paramValue==null || paramValue==""){
       return paramValue  == "" && (paramValue  = null), paramValue
    }else{
       return removeXss(paramValue);
    }
    //return paramValue  == "" && (paramValue  = null), paramValue?removeXss(paramValue):null
}

function changeURLArg(url,arg,arg_val){ 
   var pattern=arg+'=([^&]*)'; 
   var replaceText=arg+'='+arg_val; 
   if(url.match(pattern)){ 
       var tmp='/('+ arg+'=)([^&]*)/gi'; 
       tmp=url.replace(eval(tmp),replaceText); 
       return tmp; 
   }else{ 
       if(url.match('[\?]')){ 
           return url+'&'+replaceText; 
       }else{ 
           return url+'?'+replaceText; 
       } 
  } 
   return url+'\n'+arg+'\n'+arg_val;

function navhd(grbsnav,item,spantag){
	// 鼠标放在哪个li上面，span对应一道到哪里，移开后回到原位置
    var grbsnav=document.getElementById(grbsnav);
    var liArr = grbsnav.getElementsByClassName(item);
    var liWidth = liArr[0].offsetWidth;
    var span = grbsnav.getElementsByClassName(spantag)[0];
    // 计数器
    var cnt = 0;

    // for循环绑定事件
    for(var i=0; i<liArr.length; i++){
        // 自定义属性，然后绑定index属性为索引值
        liArr[i].index = i;
        // 鼠标进入事件
        liArr[i].onmouseover = function(){
            // 然span运动到该li的索引值位置
            animate(span, this.index*liWidth);
            cnt = this.index;
            animate(span, cnt*liWidth);
        }
        // 鼠标移开
        liArr[i].onmouseout = function(){
            // span运动到原位置
            animate(span, cnt*liWidth);
        }
    }
}

    // 设置超时调用
    element.timer = setInterval(function(){
        // 设置步数
        var step = (target - element.offsetLeft)/10;
        // 调整步数
        step = step > 0 ? Math.ceil(step) : Math.floor(step);
        // 设置样式
        element.style.left = element.offsetLeft + step + "px";
        // console.log(1);
        if(Math.abs(target - element.offsetLeft) < Math.abs(step)){
            element.style.left = target + "px";
            clearInterval(element.timer);
        }
    }, 20);
}

//去掉特殊字符
function checkName(val){ 
    var reg = new RegExp("[`~!@#$^&*()=|{}':;',\\[\\].<>/?~！@#￥……&*（）——|{}【】‘；：”“'。，、？]"); 
    var rs = ""; 
    for (var i = 0, l = val.length; i < val.length; i++) { 
        rs = rs + val.substr(i, 1).replace(reg, ''); 
    } 
    return rs; 
}

function newWindow(url, id) { 
	var a = document.createElement('a');

}

//加载情形数据
function loadCaseData(data){
	var m_list=[];
	if(parentcode=="0"){
		for(let i in data){
			if(data[i].PARENT_CODE==null||data[i].PARENT_CODE=="0"){
				data[i].PARENT_CODE="0";
				m_list.push(data[i])
			}
		}
	}else{
		//debugger;
		for(let i in data){
			if(data[i].PARENT_CODE==parentcode){
				m_list.push(data[i])
			}
		}
	}
	//console.log("处理情形数据");
	//console.log(m_list);
	var htmls='<ul class="ul_list" parentcode="'+parentcode+'">';
	$.each(m_list,function(i,e){
		if(e.IS_TREE_LEAF==0){
			htmls+='<li class="li_list" onclick="ThisClick(this)" code="'+e.CODE+'" parentcode="'+e.PARENT_CODE+'" is_tree_leaf="'+e.IS_TREE_LEAF+'" name="'+e.NAME+'" sourcecode="'+e.SOURCE_CODE+'" >'+
					'<i class="layui-icon layui-icon-circle-dot"></i>'+
					'<span>'+e.NAME+'</span>'+
				'</li>';
		}else{
			htmls+='<li class="li_list" onclick="ThisClick(this)" code="'+e.CODE+'" parentcode="'+e.PARENT_CODE+'" is_tree_leaf="'+e.IS_TREE_LEAF+'" name="'+e.NAME+'" sourcecode="'+e.SOURCE_CODE+'" material_code="'+e.MATERIAL_CODE+'">'+
					'<i class="layui-icon layui-icon-circle-dot"></i>'+
					'<span>'+e.NAME+'</span>'+
				'</li>';
		}
	});
	htmls+='</ul><div class="tree-list" parentcode="-1" style="display:none;"></div>';
	$('.tree-list[parentcode='+parentcode+']').html(htmls);
}

function ThisClick(c_e){
	var data=caselist;
	//选择情形		
	var m_e = {};
	m_e.CODE=$(c_e).attr("code");
	m_e.NAME=$(c_e).attr("name");
	m_e.PARENT_CODE=$(c_e).attr("parentcode");
	m_e.SOURCE_CODE=$(c_e).attr("sourcecode");
	m_e.IS_TREE_LEAF = $(c_e).attr("is_tree_leaf");
	$(c_e).parent().parent().find(".tree-list").attr("parentcode",m_e.SOURCE_CODE);
	$(c_e).parent().parent().find(".tree-list").show();
	//赋值
	material_code=$(c_e).attr("material_code");
	//console.log("===================================");
	//console.log(material_code);
	//console.log("===================================");
	nowcase = m_e;
	nowcase = m_e;
	if(nowcase.IS_TREE_LEAF=="1"){
		$("#case_submitsp").show();
	}else{
		$("#case_submitsp").hide();
	}
	$(c_e).parent().find(".li_list").removeClass("slecteact");
	$(c_e).addClass("slecteact");
	parentcode = $(c_e).attr("sourcecode");
	loadCaseData(data);
}
